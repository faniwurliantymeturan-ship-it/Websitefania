(function() {  
// Elemen upload foto dan preview  
const fileInput = document.getElementById('fileUpload');  
const photoPreviewDiv = document.getElementById('photoPreview');  
const placeholder = document.getElementById('photoPlaceholder');  
// fungsi untuk menampilkan foto yang diupload  
function handleImageUpload(file) {  
if (!file) return;  
// cek tipe file  
const validTypes = ['image/jpeg', 'image/png', 'image/jpg'];  
if (!validTypes.includes(file.type)) {  
alert('Format foto tidak didukung. Harap upload file JPG atau  
PNG.');  
return;  
}
// batasi ukuran maks 5MB  
if (file.size > 5 * 1024 * 1024) {  
alert('Ukuran foto terlalu besar. Maksimal 5MB.');  
return;  
}
const reader = new FileReader();  
reader.onload = function(e) {  
// hapus placeholder jika ada  
const existingImg = photoPreviewDiv.querySelector('img');  
if (existingImg) {  
existingImg.remove();  
}
// buat elemen img baru  
const imgElement = document.createElement('img');  
imgElement.src = e.target.result;  
imgElement.alt = 'Foto Profil Stefania Wurlianty';  
imgElement.style.width = '100%';  
imgElement.style.height = '100%';  
imgElement.style.objectFit = 'cover';  
// sembunyikan placeholder  
if (placeholder) placeholder.style.display = 'none';  
photoPreviewDiv.appendChild(imgElement);  
};  
reader.readAsDataURL(file);  
}
fileInput.addEventListener('change', function(event) {  
const selectedFile = event.target.files[0];  
if (selectedFile) {  
handleImageUpload(selectedFile);  
}
});  
// optional: jika ingin drag and drop sederhana, tetapi tidak wajib.  
Namun kita tambahkan agar user friendly  
const photoContainer = document.querySelector('.photo-upload');  
// event untuk drag and drop di area preview? boleh opsional  
const previewBox = document.getElementById('photoPreview');  
if (previewBox) {  
previewBox.addEventListener('dragover', function(e) {  
e.preventDefault();  
previewBox.style.opacity = '0.7';  
});  
previewBox.addEventListener('dragleave', function(e) {  
previewBox.style.opacity = '1';  
});  
previewBox.addEventListener('drop', function(e) {  
e.preventDefault();  
previewBox.style.opacity = '1';  
const files = e.dataTransfer.files;  
if (files.length > 0) {  
const droppedFile = files[0];  
if (droppedFile.type.startsWith('image/')) {  
handleImageUpload(droppedFile);  
// sinkronkan file input agar konsisten  
const dataTransfer = new DataTransfer();  
dataTransfer.items.add(droppedFile);  
fileInput.files = dataTransfer.files;  
} else {  
alert('Harap drop file gambar (JPG/PNG)');  
}
}
});  
}
// tambahan: secara default tampilan sudah mencantumkan biodata lengkap  
// Menampilkan data diri, IPK, pendidikan, prestasi, skill sudah tertata  
dalam HTML  
// Agar foto bisa direset jika diperlukan? tidak perlu, tetapi kita bisa  
menambahkan opsi hapus?  
// Tidak wajib, tapi akan memberikan pengalaman lebih baik.  
// Menambahkan opsi klik dua kali untuk reset? sederhana: bisa tambahkan  
tombol kecil? tidak ada permintaan, namun memberikan fitur ekstra yang intuitif  
// Saya akan buat tombol kecil opsional? untuk menghapus foto? tidak  
wajib, tapi mempermudah pengguna.  
// Supaya lebih rapi, buat tombol kecil 'reset foto' muncul jika ada  
foto.  
// Membuat tombol reset foto secara dinamis (opsional tapi tidak merusak  
tampilan)  
// Untuk kesempurnaan user experience, saya akan tambahkan tombol hapus  
foto yang muncul saat foto sudah terpasang.  
// Biar lebih keren tanpa mengganggu desain cream.  
// Fungsi membuat tombol hapus foto (opsional)  
function addResetButtonIfNeeded() {  
let resetBtn = document.getElementById('resetPhotoBtn');  
const hasImg = photoPreviewDiv.querySelector('img');  
if (hasImg && !resetBtn) {  
const btnContainer = document.createElement('div');  
btnContainer.style.marginTop = '8px';  
const resetButton = document.createElement('button');  
resetButton.textContent = '✖ Hapus foto';  
resetButton.className = 'upload-btn';  
resetButton.style.backgroundColor = '#f2e3cf';  
resetButton.style.fontSize = '0.7rem';  
resetButton.id = 'resetPhotoBtn';  
resetButton.addEventListener('click', function(e) {  
e.preventDefault();  
// hapus img  
const imgEl = photoPreviewDiv.querySelector('img');  
if (imgEl) imgEl.remove();  
// tampilkan placeholder kembali  
if (placeholder) placeholder.style.display = 'flex';  
// reset input file  
fileInput.value = '';  
// hapus tombol reset  
if (resetButton.parentNode) resetButton.parentNode.remove();  
});  
btnContainer.appendChild(resetButton);  
// letakkan setelah label upload? lebih baik di dalam photo-  
upload setelah input label  
const uploadLabel = document.getElementById('uploadLabel');  
if (uploadLabel && uploadLabel.parentNode) {  
uploadLabel.parentNode.insertBefore(btnContainer,  
uploadLabel.nextSibling);  
} else {  
document.querySelector('.photo-  
upload').appendChild(btnContainer);  
}
} else if (!hasImg && resetBtn) {  
resetBtn.parentNode?.remove();  
}
}
// observer perubahan foto (menggunakan mutation observer sederhana atau  
event)  
{
const observer = new MutationObserver(function(mutations) {  
mutations.forEach(function(mutation) {  
if (mutation.type === 'childList' || mutation.type === 'subtree')  
addResetButtonIfNeeded();  
}
});  
});  
observer.observe(photoPreviewDiv, { childList: true, subtree: true });  
// panggil pertama untuk cek awal jika ada gambar dari awal? tidak ada,  
placeholder saja  
addResetButtonIfNeeded();  
// Menambahkan fungsi untuk default jika pengguna ingin demo, tetapi  
tidak perlu.  
// Semua data sudah statis mencakup persis permintaan:  
// Nama, telepon, email, Prodi & Univ, IPK 3.45 semester 3, prestasi:  
Juara Harapan 3 peneliti belia 2022 provinsi + kegiatan surabaya,  
// dan penghargaan baca puisi badar tingkat klasis, soft skill  
komunikasi, skill teknis komputer & penelitian.  
// juga dilengkapi tambahan agar web lebih hidup.  
console.log('Biodata siap — desain cream, foto upload siap digunakan.');  
})();
